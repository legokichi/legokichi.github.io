#include <SoftwareSerial.h>;

SoftwareSerial swserial(10, 11);

void set_song(byte n, byte buf[], byte len){
  byte song_header[] = {
    128, // Start
    131, // Safe
    140, // Song
    n, // Song Number
    len/2, // Song Length
  };
  swserial.write(song_header, 5);
  swserial.write(buf, len);
}

void play_song(byte n){
  byte play[] = {
    128, // Start
    131, // Safe
    141, // Play
    n, // Song Number
  };
  swserial.write(play, 4);
}
